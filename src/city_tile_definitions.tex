% Sizes
\pgfmathsetmacro{\hexwidth}{10}
\pgfmathsetmacro{\hexedgewidth}{0.5*\hexwidth}
\pgfmathsetmacro{\abilitytextwidth}{0.5*\hexwidth-0.2}
\SQUAREROOT{3}{\solution}%
\pgfmathsetmacro{\triangleheight}{0.25*\solution*\hexwidth}

% Shapes
\def\shapeHex{
	(-0.5*\hexwidth,0) --
	(-0.25*\hexwidth,\triangleheight) --
	(0.25*\hexwidth,\triangleheight) --
	(0.5*\hexwidth,0) --
	(0.25*\hexwidth,-\triangleheight) --
	(-0.25*\hexwidth,-\triangleheight) --
	cycle
}

% Commands
\newcommand{\hextile}{
  \draw[gray] \shapeHex;
}

\newcommand{\abilitybox}[2]{
  \draw (-0.5*\hexedgewidth,-1.25) --
        (0.5*\hexedgewidth,-1.25) --
        (0.5*\hexedgewidth,-3.25) --
        (0.5*\hexedgewidth-0.15,-3.5) --
        (-0.5*\hexedgewidth+0.15,-3.5) --
        (-0.5*\hexedgewidth,-3.25) --
        cycle;
  \node [text width=\hexedgewidth cm] at (0, -1.5) {
    \begin{center}
      \color{black}{\Large \textbf #1}
    \end{center}
  };
  \node [text width=\abilitytextwidth cm] at (0, -2.5) {
    \color{black}{#2}
  };
}
\newcommand{\controlvalue}[1]{
  \draw[black, fill=white] (0.5*\hexedgewidth-0.5,-0.75) --
        (0.5*\hexedgewidth+0.5,-0.75) --
        (0.5*\hexedgewidth+0.5,-1.75) --
        (0.5*\hexedgewidth-0.5,-1.75) --
        cycle;
  \node [text width=1 cm] at (0.5*\hexedgewidth, -1.25) {
    \begin{center}
      \color{black}{\Huge \textbf #1}
    \end{center}
  };
}
\newcommand{\ringnorth}{
  \begin{scope}
    \clip \shapeHex;
    \draw (0,\triangleheight) circle (0.5cm);
  \end{scope}
}
\newcommand{\ringnorthwest}{
  \begin{scope}
    \clip \shapeHex;
    \draw (0.75*\hexedgewidth,0.5*\triangleheight) circle (0.5cm);
  \end{scope}
}
\newcommand{\ringsouthwest}{
  \begin{scope}
    \clip \shapeHex;
    \draw (0.75*\hexedgewidth,-0.5*\triangleheight) circle (0.5cm);
  \end{scope}
}
\newcommand{\ringsouth}{
  \begin{scope}
    \clip \shapeHex;
    \draw (0,-\triangleheight) circle (0.5cm);
  \end{scope}
}
\newcommand{\ringsoutheast}{
  \begin{scope}
    \clip \shapeHex;

  \end{scope}
}
\newcommand{\ringnortheast}{

  \end{scope}
}
