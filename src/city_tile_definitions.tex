\def\shapeHex{(-0.5*\hexwidth,0) -- (-0.25*\hexwidth,\triangleheight) -- (0.25*\hexwidth,\triangleheight) -- (0.5*\hexwidth,0) -- (0.25*\hexwidth,-\triangleheight) -- (-0.25*\hexwidth,-\triangleheight) -- cycle}

% Commands
\newcommand{\hextile}{

  };

  };
